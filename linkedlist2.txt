package com;

class Node{
	int data;
	Node next;
	public Node(int data) {
		this.data = data;
		this.next = null;
	}		
}
class LinkedList {
	
	Node head;
	
	public Node search(LinkedList list,int x){
		Node node = list.head;
		while(node!=null && node.data!=x){
			node=node.next;
		}		
		return node;
	}

	public static void main(String[] args) {
		LinkedList list=new LinkedList();
		
		Node node1 = new Node(5);
		Node node2 = new Node(15);
		Node node3 = new Node(52);
		Node node4 = new Node(17);
		Node node5 = new Node(29);
		
		list.head=node1;
		node1.next=node2;
		node2.next=node3;
		node3.next=node4;
		node4.next=node5;
		
		Node node_result=list.search(list,52); //returns null while input key to search is not present.
		System.out.println(node_result.data);
	}

}
