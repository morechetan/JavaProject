package com;

class Node{
	int data;
	Node next;
	public Node(int data) {
		this.data = data;
		this.next = null;
	}		
}
class LinkedList {
	
	Node head;
	
	public LinkedList() {
		this.head = null;
	}

	public Node search(int x){
		Node node = this.head;
		while(node!=null && node.data!=x){
			node=node.next;
		}		
		return node;
	}
	
	public void add(int x){
		Node temp=new Node(x);
		temp.next=this.head;
		this.head=temp;		
	}
}
